add_namespace = lakecommie

country_event = {
	id = lakecommie.1 #EVENT FOR STARRY NIGHT. YOU SHOULD CREATE TWO ADDITIONAL EVENTS
	title = lakecommie.1.t
	desc = lakecommie.1.d
	#picture = GFX_event_equestria_fail
	immediate = {log = "[GetDateText]: [Root.GetName]: event lakecommie.1"}
	is_triggered_only = yes
	option = {
		name = lakecommie.1.a
		ai_chance = {
			factor = 65
		}
		every_unit_leader = {
			limit = {
				OR = {
					has_id = 1940
					has_id = 1945
					has_id = 20041
					has_id = 2012
					has_id = 177013
				}
			}
			set_nationality = LCT
		}
		LCT = {
			if = {
				limit = { LSC = { has_war_with = LNS } }
				add_to_war = { 
				    targeted_alliance = LSC
				    enemy = LNS
				    hostility_reason = asked_to_join
				}
			}
			if = {
				limit = { LSC = { has_war_with = LSM } }
				add_to_war = { 
				    targeted_alliance = LSC
				    enemy = LSM
				    hostility_reason = asked_to_join
				}
			}
			if = {
				limit = { LSC = { has_war_with = HLQ } }
				add_to_war = { 
				    targeted_alliance = LSC
				    enemy = HLQ
				    hostility_reason = asked_to_join
				}
			}
			annex_country = { target = ROOT transfer_troops = yes }
		}
		if = {
			limit = { 
				is_ai = no
				LCT = { is_ai = yes }
			}
			custom_effect_tooltip = LCT_become_tp
			LCT = { change_tag_from = ROOT }
		}
		else = {
			custom_effect_tooltip = GAME_OVER_TT
			LCT = { country_event = lakecommie.2 }
		}
	}
	option = {
		name = lakecommie.1.b
		ai_chance = {
			factor = 35
		}
		LCT = {
			if = {
				limit = {
					LNS = { exists = yes }
				}
				create_wargoal = {	
					type = annex_everything
					target = LNS
				}
			}
			if = {
				limit = {
					LSC = { exists = yes }
				}
				create_wargoal = {	
					type = annex_everything
					target = LSC
				}
			}
			if = {
				limit = {
					LSM = { exists = yes }
				}
				create_wargoal = {	
					type = annex_everything
					target = LSM
				}
			}
		}
		LCT = { country_event = lakecommie.3 }
	}
}
